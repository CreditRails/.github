# Soroban Frontend Template (SolidJS)

A minimal dapp template that deploys a contract and generates TypeScript bindings automatically.

## Directory structure

```
├── hello-world/
│   └── contracts/hello-world/
│       ├── src/lib.rs
│       ├── Cargo.toml
│       └── Makefile
└── soroban-template-solid/
    ├── src/
    │   ├── App.tsx
    │   └── contracts/         ← generated by initialize.js
    ├── packages/              ← generated TypeScript bindings
    ├── .env
    ├── initialize.js
    ├── vite.config.ts
    └── package.json
```

## 1. Bootstrap SolidJS

```bash
npx degit solidjs/templates/vanilla/bare soroban-template-solid
cd soroban-template-solid
npm install
npm install dotenv glob util
```

## 3. `initialize.js` (complete)

```js
import "dotenv/config";
import { mkdirSync, writeFileSync, rmSync } from "fs";
import { execSync, exec } from "child_process";
import path from "path";
import { fileURLToPath } from "url";
import { sync as glob } from "glob";
import { promisify } from "util";

const __filename = fileURLToPath(import.meta.url);
const dirname = path.dirname(__filename);

var smartContracts = Array();
const execAsync = promisify(exec);

// ── Create user ──────────────────────────────────────────────────────────────
function createUser() {
  execSync(`stellar keys generate --fund ${process.env.PUBLIC_STELLAR_ACCOUNT} | true`);
}

// ── Build contracts ──────────────────────────────────────────────────────────
function removeFiles(pattern) {
  glob(pattern).forEach((entry) => rmSync(entry));
}

function buildAll() {
  removeFiles(`${dirname}/target/wasm32v1-none/release/*.wasm`);
  removeFiles(`${dirname}/target/wasm32v1-none/release/*.d`);
  execSync(`stellar contract build`);
  console.log("Build complete");
}

// ── Deploy contracts ─────────────────────────────────────────────────────────
function filenameNoExtension(filename) {
  return path.basename(filename, path.extname(filename));
}

async function deploy(wasm) {
  const { stdout } = await execAsync(
    `stellar contract deploy --wasm ${wasm} --ignore-checks ` +
    `--alias ${filenameNoExtension(wasm)} ` +
    `--source ${process.env.PUBLIC_STELLAR_ACCOUNT} ` +
    `--network ${process.env.PUBLIC_STELLAR_NETWORK} ` +
    `--rpc-url ${process.env.PUBLIC_STELLAR_RPC_URL} ` +
    `--network-passphrase "${process.env.PUBLIC_STELLAR_NETWORK_PASSPHRASE}"`,
  );
  smartContracts.push({
    alias: filenameNoExtension(wasm),
    wasm,
    contractid: stdout.substring(0, stdout.length - 1),
  });
  console.log(`Deployed ${filenameNoExtension(wasm)}`);
}

async function deployAll() {
  const wasmFiles = glob(`${dirname}/target/wasm32v1-none/release/*.wasm`);
  for (const wasm of wasmFiles) await deploy(wasm);
}

// ── Create TypeScript bindings ───────────────────────────────────────────────
function bind({ alias, contractid }) {
  execSync(
    `stellar contract bindings typescript --contract-id ${contractid} ` +
    `--output-dir ${dirname}/packages/${alias} --overwrite`,
  );
  execSync(`(cd ${dirname}/packages/${alias} && npm i && npm run build)`);
  execSync(`pnpm add file:./packages/${alias}`);
}

async function bindAll() {
  for (const contract of smartContracts) await bind(contract);
}

// ── Import bindings ──────────────────────────────────────────────────────────
function importContract({ alias, contractid }) {
  const outputDir = `${dirname}/src/contracts/`;
  mkdirSync(outputDir, { recursive: true });

  const content =
    `import * as Client from '${alias}';\n` +
    `export default new Client.Client({\n` +
    `  contractId: "${contractid}",\n` +
    `  networkPassphrase: "${process.env.PUBLIC_STELLAR_NETWORK_PASSPHRASE}",\n` +
    `  rpcUrl: "${process.env.PUBLIC_STELLAR_RPC_URL}",\n` +
    (process.env.PUBLIC_STELLAR_NETWORK === "local" ? `  allowHttp: true,\n` : "") +
    `});\n`;

  writeFileSync(`${outputDir}/${alias}.ts`, content);
  console.log(`Created import for ${alias}`);
}

function importAll() {
  smartContracts.forEach(importContract);
}

// ── Main ─────────────────────────────────────────────────────────────────────
async function main() {
  createUser();
  buildAll();
  await deployAll();
  await bindAll();
  importAll();
}

main().catch((e) => {
  console.error("Initialization failed", e);
  process.exit(1);
});
```

## 5. `src/App.tsx` — calling the contract

```tsx
import type { Component } from 'solid-js';
import { createSignal } from "solid-js";
import helloWorld from './contracts/hello_world';

const App: Component = () => {
  const [input, setInput] = createSignal('');
  const [greeting, setGreeting] = createSignal('');
  const [loading, setLoading] = createSignal(false);
  const [error, setError] = createSignal<string | null>(null);

  async function getGreeting(e?: Event) {
    e?.preventDefault();
    setError(null);
    setLoading(true);
    try {
      const { result } = await helloWorld.hello({ to: input() || 'you' });
      setGreeting(Array.isArray(result) ? result.join(' ') : String(result));
    } catch (err: any) {
      setError(err?.message || 'Unknown error');
    } finally {
      setLoading(false);
    }
  }

  return (
    <div>
      <h1>Hello Soroban!</h1>
      <form onSubmit={getGreeting}>
        <input
          type="text"
          placeholder="Type your name..."
          value={input()}
          onInput={(e: any) => setInput(e.target.value)}
        />
        <button type="submit" disabled={loading()}>
          {loading() ? 'Sending...' : 'Send'}
        </button>
      </form>
      {error() && <p style={{ color: 'red' }}>Error: {error()}</p>}
      <p>{greeting()}</p>
    </div>
  );
};

export default App;
```

## 6. Run

```bash
node initialize.js   # deploy + bind once
npm run dev          # http://localhost:3000
```

## How `initialize.js` works end-to-end

| Step | Function | What it does |
|------|----------|--------------|
| 1 | `createUser()` | `stellar keys generate --fund <account>` |
| 2 | `buildAll()` | `stellar contract build`, cleans old Wasms first |
| 3 | `deployAll()` | Deploys each `.wasm`, collects contract IDs |
| 4 | `bindAll()` | `stellar contract bindings typescript` → builds npm package |
| 5 | `importAll()` | Writes `src/contracts/<alias>.ts` with configured Client |
